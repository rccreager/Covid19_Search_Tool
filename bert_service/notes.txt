start server like so:

# keep track of the following:
## gateway IP address (e.g. 1.2.3.4)
## subnet mask (e.g. /11)
## server container ID
## client container ID

# first start server
docker run -it -runtime nvidia covid19-search-tool:29apr_1721_bert 4 40
# then main
docker run -it covid19-search-tool:29apr_1610_main

#create network
docker network create my-net --gateway=1.2.3.4 --subnet=1.2.3.4/11
# add server to network with unique ip address
docker network connect --ip 1.2.4.8 my-net 5b68b9a6f10b 
# add client to network
docker network connect my-net 9f4ee37c8424
